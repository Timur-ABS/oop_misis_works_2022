#include <iostream>
#include <vector>
#include<set>
#include<string>
#include<algorithm>
struct st {
	int n;
	int i;
};
bool cmp(st x, st y) {
	if (x.n < y.n) {
		return true;
	}
	else if (x.n == y.n) {
		return x.i > y.i;
	}
	else {
		return false;
	}
}
int main() {
	int i, m,n;
	std::cin >> n;
	std::vector <st> A(n);
	std::vector <int> B(n);
	for (int i = 0; i < n; ++i) {
		std::cin >> A[i].n;
		A[i].i=i;
	}
	sort(A.begin(), A.end(), cmp);
	int ans = A[0].n;
	m = A[0].i;
	for (int i = 1; i < n; ++i) {
		if (A[i - 1].n != A[i].n) {
			if (A[i].i < m) {
				m = A[i].i;
				ans = A[i].n;
			}
		}
	}
	std::cout << ans;
}
