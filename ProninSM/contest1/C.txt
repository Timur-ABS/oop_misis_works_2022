#include <iostream>
#include <vector>
#include<set>
#include<string>
#include<algorithm>
struct st {
	int s;
	int i;
};
bool cmp(st x, st y) {
	if (x.s > y.s) {
		return true;
	}
	else if (x.s == y.s) {
		return x.i < y.i;
	}
	else {
		return false;
	}
}
int main() {
	int i, m,n;
	std::cin >> n;
	std::vector <st> A(n);
	std::vector <int> B(n);
	for (int i = 0; i < n; ++i) {
		int x1, x2, x3, x4;
		std::cin >> x1 >> x2 >> x3 >> x4;
		A[i].s=x1+x2+x3+x4;
		A[i].i=i;
	}
	sort(A.begin(), A.end(), cmp);
	for (int i = 0; i < n; ++i) {
		if (A[i].i == 0) {
			std::cout << i + 1;
			break;
		}
	}
}
